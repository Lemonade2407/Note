- controller
```
//获取文件的形参为MultipartFile类
//本地存储
public Result handleFileUpload(MultipartFile file)
{
	//获取原始文件名
	String originalFilename = file.getOriginalFilename();
	//生成唯一文件名
	String extension = originalFilename.substring(originalFilename.lastIndexOf("."));//获取后缀
	String newFileName = UUID.randmUUID().toString()+extension;//随机UUID文件名
	//保存文件
	file.transferTo(new File("path"))
}

//阿里云存储
public Result handleFileUpload(MultipartFile file)
{
	String url = aliyunOSSOperator.upload(file.getBytes(),file.getOriginalFilename)
}
```
